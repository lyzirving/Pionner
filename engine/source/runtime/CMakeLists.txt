message("runtime dir: ${CMAKE_CURRENT_SOURCE_DIR}")

set(TARGET_NAME Engine)

file(GLOB ENGINE_HEADERS CONFIGURE_DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/*.h)
file(GLOB ENGINE_SOURCES CONFIGURE_DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)

add_library( ${TARGET_NAME} ${ENGINE_HEADERS} ${ENGINE_SOURCES} )

set(ENGINE_INCLUDE_DIR 
    ${CMAKE_CURRENT_SOURCE_DIR}
)

target_include_directories(
  ${TARGET_NAME} 
  PUBLIC ${ENGINE_INCLUDE_DIR}
)
